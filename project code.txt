#include <iostream>
#include <string>
using namespace std;

struct Task {
    int id;
    string description;
    bool completed;
    Task* next;
};

Task* head = NULL;

void addTask(int id, string desc) {
    Task* newTask = new Task();
    newTask->id = id;
    newTask->description = desc;
    newTask->completed = false;
    newTask->next = NULL;

    if (head == NULL) {
        head = newTask;
    } else {
        Task* temp = head;
        while (temp->next != NULL) {
            temp = temp->next;
        }
        temp->next = newTask;
    }
    cout << "Task added successfully!\n";
}

void viewTasks() {
    Task* temp = head;
    if (temp == NULL) {
        cout << "No tasks found!\n";
        return;
    }
    while (temp != NULL) {
        cout << "ID: " << temp->id << " | Desc: " << temp->description << " | ";
        cout << (temp->completed ? "Completed" : "Pending") << endl;
        temp = temp->next;
    }
}

void markCompleted(int id) {
    Task* temp = head;
    while (temp != NULL) {
        if (temp->id == id) {
            temp->completed = true;
            cout << "Task marked as completed.\n";
            return;
        }
        temp = temp->next;
    }
    cout << "Task not found!\n";
}

void deleteTask(int id) {
    Task* temp = head;
    Task* prev = NULL;

    while (temp != NULL && temp->id != id) {
        prev = temp;
        temp = temp->next;
    }

    if (temp == NULL) {
        cout << "Task not found!\n";
        return;
    }

    if (prev == NULL) {
        head = temp->next;
    } else {
        prev->next = temp->next;
    }
    delete temp;
    cout << "Task deleted.\n";
}

int main() {
    int choice, id;
    string desc;

    do {
        cout << "\n--- TO-DO LIST MANAGER ---\n";
        cout << "1. Add Task\n2. View Tasks\n3. Mark Task Completed\n4. Delete Task\n5. Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                cout << "Enter Task ID: ";
                cin >> id;
                cin.ignore();
                cout << "Enter Task Description: ";
                getline(cin, desc);
                addTask(id, desc);
                break;
            case 2:
                viewTasks();
                break;
            case 3:
                cout << "Enter Task ID to mark complete: ";
                cin >> id;
                markCompleted(id);
                break;
            case 4:
                cout << "Enter Task ID to delete: ";
                cin >> id;
                deleteTask(id);
                break;
            case 5:
                cout << "Goodbye!\n";
                break;
            default:
                cout << "Invalid choice!\n";
        }

    } while (choice != 5);

    return 0;
}
